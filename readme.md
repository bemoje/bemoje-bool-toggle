## Installation

```sh
npm install @bemoje/bool-toggle
npm install --save @bemoje/bool-toggle
npm install --save-dev @bemoje/bool-toggle
```

## Usage

```javascript

import boolToggle from '@bemoje/bool-toggle'

boolToggle(true)
//=> false

boolToggle(false)
//=> true

let bool = false

bool = boolToggle(bool)

bool === true
//=> true

bool = boolToggle(bool)

bool === false
//=> true

```
